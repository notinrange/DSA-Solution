class Solution {
public:
    int characterReplacement(string s, int k) {
      unordered_map<char,int> count;
        int res = 0,l=0,r,maxf=0;
        for( r =0;r<s.size();r++){
            count[s[r]]=1+count[s[r]];
            maxf = max(maxf,count[s[r]]);
            while(((r-l+1)-maxf)>k){
                count[s[l]]-=1;
                l+=1;
            }
            res = max(res,r-l+1);
        }
        return res;
    }
};
